#Programa para ilustrar o uso de if...else if

{
<< "Digite um numero: ";
>> a;

?a > 5? -> 1;
    ?a < 5? -> 2;       #se a <= 5, verifica se a < 5
        << "o numero eh igual a 5 ";
        << 0; -> 3;     # se a = 5, imprime 0
    2;
        << "o numero eh menor que 5 ";
        << -1; -> 3;    #se a < 5, imprime -1 e continua o programa
1;
    << "o numero eh maior que 5 ";
    << 1;               #se a > 5, imprime 1 e continua o programa
3;

#contuacao do programa
<< "\n", 497;
}
